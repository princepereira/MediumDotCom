void sort(int[] arr , int l, int r){
    if (l < r){
      int m = (r + l)/2;
      sort(arr, l, m);
      sort(arr, m+1, r);
      merge(arr, l, m, r);
    }
  }
void merge(int[] arr, int l, int m, int r){
    int n1 = m - l +1;
    int n2 = r - m;
    int[] L = new int[n1];
    int[] R = new int[n2];
   System.arraycopy(arr, l, L, 0, n1);
   System.arraycopy(arr, m+1, R, 0, n2);
   int i = 0 , j = 0;
   while(i < n1 && j < n2){
      if(L[i] <= R[j]){
        arr[l++] = L[i++];
      } else {
        arr[l++] = R[j++];
      }
   }
   while(i < n1){
      arr[l++] = L[i++];
   }
   while(j < n2){
      arr[l++] = R[j++];
   }
    
  }
}

package main
import (
   "fmt"
)
func main() {
   arr := []int{56, 23, 12, 8, 34, 14}
   fmt.Println("Before : ", arr)
   sort(arr)
   fmt.Println("After  : ", arr)
}
func sort(arr []int) {
   if len(arr) <= 1 {
       return
   }
   mergeSort(arr, 0, len(arr)-1)
}
func mergeSort(arr []int, l, r int) {
   if l < r {
      m := (l + r) / 2
      mergeSort(arr, l, m)
      mergeSort(arr, m+1, r)
      merge(arr, l, m, r)
   }
}
func merge(arr []int, l, m, r int) {
   L := make([]int, m-l+1)
   R := make([]int, r-m)
   copy(L, arr[l:m+1])
   copy(R, arr[m+1:r+1])
   i, j := 0, 0
   for i < len(L) && j < len(R) {
       if L[i] < R[j] {
           arr[l] = L[i]
           i++
       } else {
           arr[l] = R[j]
           j++
       }
       l++
   }
   for i < len(L) {
       arr[l] = L[i]
       i++
       l++
   }
   for j < len(R) {
       arr[l] = R[j]
       j++
       l++
   }
}
