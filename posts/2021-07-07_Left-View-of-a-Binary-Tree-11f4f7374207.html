Left View of a Binary Tree

Use level order traversal for this.

Program in Golang

package main
import (
 "fmt"
)
type node struct {
 data  int
 left  *node
 right *node
}
func newNode(data int) *node {
 n := new(node)
 n.data = data
 return n
}
func leftTreeView(n *node) {
 var q []*node
 q = append(q, n)
 for len(q) > 0 {
   qSize := len(q)
   for i := 0; i < qSize; i++ {
     t := q[0]
     if i == 0 {
       fmt.Print(t.data, " ")
     }
     q = q[1:]
     if t.left != nil {
       q = append(q, t.left)
     }
     if t.right != nil {
       q = append(q, t.right)
     }
   }
 }
 fmt.Println()
}
func main() {
 fmt.Println("Left view of the binary tree is :")
 root := newNode(1)
 root.left = newNode(2)
 root.right = newNode(6)
 root.left.right = newNode(3)
 root.right.right = newNode(7)
 root.right.right.left = newNode(4)
 root.right.right.right = newNode(8)
 root.right.right.left.left = newNode(5)
 leftTreeView(root)
}

Output :

Left view of the binary tree is :
1 2 3 4 5
