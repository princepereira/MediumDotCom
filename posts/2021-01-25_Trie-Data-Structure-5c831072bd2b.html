import java.util.*;
public class Trie{
private class TrieNode{
    Map<Character, TrieNode> child;
    boolean endOfWord;
    public TrieNode(){
      child = new HashMap<>();
      endOfWord = false;
    }
  }
  private final TrieNode root;
  public Trie(){
    root = new TrieNode();
  }
  public void insert(String word){
    insertRecursive(root, word, 0);
  }
  private void insertRecursive(TrieNode curNode, String word, int index){
    if (index == word.length()){
      curNode.endOfWord = true;
      return;
    }
    char ch = word.charAt(index);
    TrieNode node = curNode.child.get(ch);
    if (node == null){
      node = new TrieNode();
      curNode.child.put(ch, node);
    }
    insertRecursive(node, word, index+1);
  }
  public boolean search(String word){
    return searchRecursive(root, word, 0);
  }
  private boolean searchRecursive(TrieNode curNode, String word, int index){
    if (index == word.length()){
      return curNode.endOfWord;
    }
    char ch = word.charAt(index);
    TrieNode node = curNode.child.get(ch);
    if (node == null){
      return false;
    }
    return searchRecursive(node, word, index+1);
  }
  public void delete(String word){
   deleteRecursive(root, word, 0); 
  }
  private boolean deleteRecursive(TrieNode curNode, String word, int index){
    if (index == word.length()){
      if (!curNode.endOfWord){
        return false;
      }
      curNode.endOfWord = false;
      return curNode.child.size() == 0;
    }
    char c = word.charAt(index);
    TrieNode node = curNode.child.get(c);
    if (node == null){
      return false;
    }
    boolean shouldDelCurNode = deleteRecursive(node, word, index+1);
    if (shouldDelCurNode){
      curNode.child.remove(c);
      return curNode.child.size() == 0;
    }
    return false;
  }
  public static void main(String[] args){
    Trie o = new Trie();
    o.insert("beautiful");
    o.insert("hello");
    o.insert("awesome");
    System.out.println("beautiful present "+o.search("beautiful")); 
    System.out.println("beautifull present "+o.search("beautifull")); 
    System.out.println("awesome present "+o.search("awesome")); 
    o.delete("awesome");
    System.out.println("After deleting awesome : "+o.search("awesome"));
  }
}
