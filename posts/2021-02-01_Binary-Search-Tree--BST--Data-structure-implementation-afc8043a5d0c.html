Binary Search Tree (BST) Data structure implementation

  • Tree data structure
  • Each node has at most 2 child nodes
  • Right sub tree is greater than Left sub tree

  private void insert(Node current, int val){

      if (current.right == null) {
        Node node = new Node(val);
        current.right = node;
      } else {
        insert(current.right, val);
      }
    }
  }

  }
  private Node delete(Node current, int val){
    if (current == null){
      return null;
    }
    if (current.data > val){
      current.left = delete(current.left, val);
    } else if (current.data < val){
      current.right = delete(current.right, val);
    } else {
      if (current.left == null){
        return current.right;
      } else if (current.right == null){
        return current.left;
      }
      current.data = minKey(current.right);
      current.right = delete(current.right, current.data);
    }
    return current;
  }

Entire Program

import java.util.*;
public class Tree{
  private class Node {
    int data;
    Node left;
    Node right;
  public Node(int val){
      this.data = val;
      this.left = null;
      this.right = null;
    }
  }
  private Node root;
  public void insert(int val){
    if (root == null){
      root = new Node(val);
      return;
    }
    insert(root, val);
  }
  private void insert(Node current, int val){
    if (current.data == val){
      return;
    }
    if (current.data > val) {
      if (current.left == null) {
        Node node = new Node(val);
        current.left = node;
      } else {
        insert(current.left, val);
      }
    } else {
      if (current.right == null) {
        Node node = new Node(val);
        current.right = node;
      } else {
        insert(current.right, val);
      }
    }
  }
  public void inorder(){
    if (root != null){
      System.out.println("\nInorder");
      inorder(root);
    }
    System.out.println();
  }
  private void inorder(Node current){
    if (current == null){
      return;
    }
    inorder(current.left);
    System.out.print(current.data + " ");
    inorder(current.right);
  }
  public void preorder(){
    if (root != null){
      System.out.println("\nPreorder");
      preorder(root);
    }
    System.out.println();
  }
  private void preorder(Node current){
    if (current == null){
      return;
    }
    System.out.print(current.data + " ");
    preorder(current.left);
    preorder(current.right);
  }
  public void postorder(){
    if (root != null){
      System.out.println("\nPostorder");
      postorder(root);
    }
    System.out.println();
  }
  private void postorder(Node current){
    if (current == null){
      return;
    }
    System.out.print(current.data + " ");
    postorder(current.left);
    postorder(current.right);
  }
  public void delete(int val){
    System.out.printf("\nDelete %d \n",val);
    if (root != null){
      root = delete(root, val);
    }
  }
  private Node delete(Node current, int val){
    if (current == null){
      return null;
    }
    if (current.data > val){
      current.left = delete(current.left, val);
    } else if (current.data < val){
      current.right = delete(current.right, val);
    } else {
      if (current.left == null){
        return current.right;
      } else if (current.right == null){
        return current.left;
      }
      current.data = minKey(current.right);
      current.right = delete(current.right, current.data);
    }
    return current;
  }
  private int minKey(Node current){
    int minV = current.data;
    while (current.left != null){
      minV = current.left.data;
      current = current.left;
    }
    return minV;
  }
  public static void main(String[] args){
    Tree t = new Tree();
    t.insert(120);
    t.insert(50);
    t.insert(150);
    t.insert(500);
    t.insert(30);
    t.insert(80);
    t.inorder();
    t.preorder();
    t.postorder();
    t.delete(150);
    t.inorder();
    System.out.println();
  }
}
